26# URL patterns for cleaning (US1) 

27MARKDOWN_LINK_PATTERN = re.compile(r"\[([^\]]+)\]\([^)]+\)") 

28# Match bare URLs but stop before Japanese characters, parentheses, and brackets 

29BARE_URL_PATTERN = re.compile(r"https?://[^\s\u3000-\u9fff\uff00-\uffef）」』】\]]+") 

30# Match www. URLs (e.g., www.example.com) 

31WWW_URL_PATTERN = re.compile(r"www\.[^\s\u3000-\u9fff\uff00-\uffef）」』】\]]+") 

32URL_TEXT_PATTERN = re.compile(r"^(?:https?://|www\.)")

53# ISBN-13: 978/979 + 10 digits with optional hyphens 

54# ISBN-10: 10 digits/chars with optional hyphens (last char can be X) 

55ISBN_PATTERN = re.compile(

57 r"(?:" 

58 r"97[89][-\s]?\d[-\s]?\d{1,5}[-\s]?\d{1,7}[-\s]?\d" # ISBN-13 with hyphens 

59 r"|97[89]\d{10}" # ISBN-13 without hyphens 

60 r"|\d[-\s]?\d{1,5}[-\s]?\d{1,7}[-\s]?[\dXx]" # ISBN-10 with hyphens 

61 r"|\d{9}[\dXx]" # ISBN-10 without hyphens 

62 r")"

65# ISBN with context pattern (US3 Phase 4) 

66# Matches ISBNs with brackets and labels: 

67# - （ISBN: 978-...） or (ISBN: 978-...) 

68# - ISBN: 978-... or ISBN-10: ... or ISBN-13: ... 

69# - （978-...） (ISBN without label in brackets) - only for ISBN-13 starting with 978/979 

70ISBN_WITH_CONTEXT_PATTERN = re.compile( 

71 r"(?:" 

72 # Pattern 1: Bracketed ISBN with optional label 

73 r"[（(]" # Opening bracket (full-width or half-width) 

74 r"(?:[Ii][Ss][Bb][Nn](?:-1[03])?[\s:：-]*)?" # Optional ISBN/ISBN-10/ISBN-13 label 

75 r"(?:" 

76 r"97[89][-\s]?\d[-\s]?\d{1,5}[-\s]?\d{1,7}[-\s]?\d" # ISBN-13 with hyphens 

77 r"|97[89]\d{10}" # ISBN-13 without hyphens 

78 r"|\d[-\s]?\d{1,5}[-\s]?\d{1,7}[-\s]?[\dXx]" # ISBN-10 with hyphens 

79 r"|\d{9}[\dXx]" # ISBN-10 without hyphens 

80 r")" 

81 r"[）)]" # Closing bracket (must match opening) 

82 r"|" 

83 # Pattern 2: ISBN with label (no brackets required) 

84 r"[Ii][Ss][Bb][Nn](?:-1[03])?[\s:：-]+" # ISBN/ISBN-10/ISBN-13 label (required) 

85 r"(?:" 

86 r"97[89][-\s]?\d[-\s]?\d{1,5}[-\s]?\d{1,7}[-\s]?\d" # ISBN-13 with hyphens 

87 r"|97[89]\d{10}" # ISBN-13 without hyphens 

88 r"|\d[-\s]?\d{1,5}[-\s]?\d{1,7}[-\s]?[\dXx]" # ISBN-10 with hyphens 

89 r"|\d{9}[\dXx]" # ISBN-10 without hyphens 

90 r")" 

91 r")" 

92)

95# Matches brackets containing only ASCII letters, numbers, spaces, hyphens, periods, commas

97PAREN_ENGLISH_FULL = re.compile(r"（[A-Za-z][A-Za-z0-9\s\-.,]*）")

140def split_into_pages(markdown: str) -> list[Page]: 

141 """Split book.md into pages based on page markers.""" 

142 pattern = r"^--- Page (\d+) \(page_\d+\.png\) ---$" 

143 parts = re.split(pattern, markdown, flags=re.MULTILINE) 

144 

145 pages = [] 

146 # parts[0] is text before first marker (usually empty) 

147 # parts[1], parts[2] = page_number, page_content 

148 # parts[3], parts[4] = page_number, page_content ... 

149 for i in range(1, len(parts), 2): 

150 page_num = int(parts[i]) 

151 raw_text = parts[i + 1] if i + 1 < len(parts) else "" 

152 cleaned = clean_page_text(raw_text) 

153 if cleaned.strip(): 

154 pages.append(Page(number=page_num, text=cleaned)) 

155 return pages

158def _clean_urls(text: str) -> str: 

159 """Replace URLs with 'ウェブサイト' for TTS. 

160 

161 - Markdown links: Keep link text, remove URL 

162 - URL as link text: Replace with 'ウェブサイト' 

163 - Bare URLs (http(s)://...): Replace with 'ウェブサイト' 

164 - www. URLs: Replace with 'ウェブサイト' 

165 """ 

166 

167 # Step 1: Handle Markdown links 

168 def replace_markdown_link(match: re.Match[str]) -> str: 

169 link_text = match.group(1) 

170 # If link text is a URL, replace with 'ウェブサイト' 

171 if URL_TEXT_PATTERN.match(link_text): 

172 return "ウェブサイト" 

173 return link_text 

174 

175 text = MARKDOWN_LINK_PATTERN.sub(replace_markdown_link, text) 

176 

177 # Step 2: Replace bare http(s):// URLs with 'ウェブサイト' 

178 text = BARE_URL_PATTERN.sub("ウェブサイト", text) 

179 

180 # Step 3: Replace www. URLs with 'ウェブサイト' 

181 text = WWW_URL_PATTERN.sub("ウェブサイト", text) 

182 

183 return text

223def _clean_isbn(text: str) -> str: 

224 """Remove ISBN numbers from text for TTS. 

225 

226 Removes all ISBN patterns including: 

227 - Parenthetical ISBNs: （ISBN: 978-...） → removed with brackets 

228 - Labeled ISBNs: ISBN: 978-... → removed with label 

229 - Bare ISBNs: ISBN978-... → removed 

230 

231 Also normalizes spaces after removal: 

232 - Full-width spaces (　) → removed completely 

233 - Exactly 2 consecutive half-width spaces → removed completely 

234 - 3+ consecutive half-width spaces → collapsed to single space 

235 

236 Note: Does not strip leading/trailing spaces to preserve text structure. 

237 Whitespace-only input is preserved as-is. 

238 """ 

239 # Preserve whitespace-only input unchanged 

240 if not text.strip(): 

241 return text 

242 

243 # Step 1: Remove ISBNs with context (brackets, labels) 

244 text = ISBN_WITH_CONTEXT_PATTERN.sub("", text) 

245 

246 # Step 2: Remove any remaining bare ISBNs 

247 text = ISBN_PATTERN.sub("", text) 

248 

249 # Step 3: Normalize spaces 

250 # First, remove all full-width spaces 

251 text = text.replace("\u3000", "") 

252 # Then, handle half-width spaces: 

253 # - 3+ consecutive spaces → single space 

254 text = re.sub(r" {3,}", " ", text) 

255 # - Exactly 2 consecutive spaces → no space 

256 text = text.replace(" ", "") 

257 

258 return text

263

266 - 半角括弧内の英語のみ: トイル(Toil) → トイル

270 - 空括弧: （）→ 保持 

271 - 数字のみ括弧: （1.0）→ 保持 

272 

273 Args: 

274 text: Input text containing parenthetical terms 

275 

276 Returns: 

277 Text with English-only parenthetical terms removed 

278 """ 

279 text = PAREN_ENGLISH_FULL.sub("", text) 

280 text = PAREN_ENGLISH_HALF.sub("", text)

284def clean_page_text(text: str, heading_marker: str | None = None) -> str: 

285 """Clean a single page's text for TTS consumption. 

286 

287 Args: 

288 text: Text to clean 

289 heading_marker: Optional marker to preserve through cleaning (e.g., HEADING_MARKER) 

290 """ 

291 # Preserve heading marker through text cleaning (MeCab strips Unicode private use area) 

292 if heading_marker and heading_marker in text: 

293 text = text.replace(heading_marker, _HEADING_PLACEHOLDER) 

294 

295 # NEW: Text cleaning for TTS (before markdown cleanup) 

296 # Process in specific order to avoid interference 

297 text = _clean_urls(text) # US1: Remove URLs 

298 text = _clean_isbn(text) # US4: Remove ISBN 

299 text = _clean_number_prefix(text) # US2: No.X → ナンバーX 

300 text = _clean_chapter(text) # US2: Chapter X → 第X章 

301 text = _clean_parenthetical_english(text) # US5: Remove (English) 

302 text = _normalize_references(text) # US2/3: 図X.Y → ずXのY 

303 

304 # Remove HTML comments (figure markers) 

305 text = HTML_COMMENT_PATTERN.sub("", text) 

306 

307 # Remove figure description paragraphs (lines starting with 図は、) 

308 text = FIGURE_DESC_PATTERN.sub("", text) 

309 

310 # Remove page number markers like "1 / 1", "はじめに 1 / 3" 

311 text = PAGE_NUMBER_PATTERN.sub("", text) 

312 

313 # Remove markdown heading markers but keep text 

314 text = HEADING_PATTERN.sub("", text) 

315 

316 # Remove bold/italic markers 

317 text = BOLD_ITALIC_PATTERN.sub(r"\1", text) 

318 

319 # Remove horizontal rules 

320 text = HORIZONTAL_RULE_PATTERN.sub("", text) 

321 

322 # Convert consecutive list items into single line with periods 

323 def convert_list_block(match: re.Match) -> str: 

324 lines = match.group(0).strip().split("\n") 

325 items = [] 

326 for line in lines: 

327 content = LIST_MARKER_PATTERN.sub("", line) 

328 if content and content[-1] not in "。！？、": 

329 content += "。" 

330 items.append(content) 

331 return "".join(items) 

332 

333 text = LIST_BLOCK_PATTERN.sub(convert_list_block, text) 

334 

335 # Remove inline code backticks 

336 text = INLINE_CODE_PATTERN.sub(r"\1", text) 

337 

338 # Remove code blocks 

339 text = CODE_BLOCK_PATTERN.sub("", text) 

340 

341 # Remove trailing whitespace from lines 

342 text = TRAILING_WHITESPACE_PATTERN.sub("", text) 

343 

344 # Remove markdown line breaks (trailing double space) 

345 text = DOUBLE_SPACE_PATTERN.sub("", text) 

346 

347 # Collapse multiple blank lines into one 

348 text = MULTIPLE_NEWLINES_PATTERN.sub("\n\n", text) 

349 

350 # Apply TTS normalization and reading rules 

351 # 0. Punctuation normalization (add commas for natural reading) 

352 text = normalize_punctuation(text) 

353 # 1. Number normalization (123 → ひゃくにじゅうさん) 

354 text = normalize_numbers(text) 

355 # 2. Static dictionary (critical terms like SRE, API, AWS) 

356 text = apply_reading_rules(text) 

357 # 3. LLM-generated dictionary (additional terms) 

358 if _LLM_READINGS: 

359 text = apply_llm_readings(text, _LLM_READINGS) 

360 # 4. MeCab: Convert remaining kanji to kana 

361 if ENABLE_KANJI_CONVERSION: 

362 text = convert_to_kana(text) 

363 

364 # Restore heading marker if it was preserved 

365 if heading_marker and _HEADING_PLACEHOLDER in text: 

366 text = text.replace(_HEADING_PLACEHOLDER, heading_marker) 

367 

368 return text.strip()

371def split_text_into_chunks(text: str, max_chars: int = 500) -> list[str]: 

372 """Split text into chunks suitable for TTS generation. 

373 

374 Splits on sentence boundaries (。！？) to maintain natural reading flow. 

375 """ 

376 if len(text) <= max_chars: 

377 return [text] if text.strip() else [] 

378 

379 chunks = [] 

380 remaining = text 

381 

382 while remaining: 

383 if len(remaining) <= max_chars: 

384 if remaining.strip(): 

385 chunks.append(remaining.strip()) 

386 break 

387 

388 # Find the best split point within max_chars 

389 split_pos = _find_split_position(remaining, max_chars) 

390 chunk = remaining[:split_pos].strip() 

391 if chunk: 

392 chunks.append(chunk) 

393 remaining = remaining[split_pos:].strip() 

394 

395 return chunks

398def _find_split_position(text: str, max_chars: int) -> int: 

399 """Find the best position to split text, preferring sentence boundaries.""" 

400 search_region = text[:max_chars] 

401 

402 # Try Japanese sentence endings: 。！？ 

403 for delimiter in ["。", "！", "？", "!", "?", "\n\n", "\n", "、", ","]: 

404 pos = search_region.rfind(delimiter) 

405 if pos > max_chars // 3: # Don't split too early 

406 return pos + len(delimiter) 

407 

408 # Fallback: split at max_chars 

409 return max_chars
