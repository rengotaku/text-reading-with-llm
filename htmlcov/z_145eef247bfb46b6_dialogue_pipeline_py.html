56def apply_readings_to_text(text: str) -> str: 

57 """Apply reading rules and dictionary to text for TTS. 

58 

59 Applies in order: 

60 1. Number normalization (123 → ひゃくにじゅうさん) 

61 2. Static reading rules (common technical terms) 

62 3. LLM-generated readings (per-book dictionary) 

63 

64 Args: 

65 text: Input text to process 

66 

67 Returns: 

68 Text with readings applied 

69 """ 

70 # 1. Normalize numbers (must be first - before reading rules) 

71 text = normalize_numbers(text) 

72 

73 # 2. Apply static reading rules (critical terms like SRE, API, AWS) 

74 text = apply_reading_rules(text) 

75 

76 # 3. Apply LLM-generated readings if available 

77 if _READINGS: 

78 text = apply_llm_readings(text, _READINGS) 

79 

80 return text

124def parse_dialogue_xml(xml_str_or_path: str) -> list[dict[str, Any]]: 

125 """対話形式XMLを解析してセクションリストを返す. 

126 

127 XMLの文字列またはファイルパスを受け取り、各セクションの情報を 

128 dictのリストとして返す。 

129 

130 Args: 

131 xml_str_or_path: XML文字列またはファイルパス文字列 

132 

133 Returns: 

134 セクション情報のリスト。各dictは以下のキーを持つ: 

135 - section_number: セクション番号 (str) 

136 - section_title: セクションタイトル (str) 

137 - introduction: {"speaker": str, "text": str} または None 

138 - utterances: [{"speaker": str, "text": str}, ...] 

139 - conclusion: {"speaker": str, "text": str} または None 

140 

141 Raises: 

142 ValueError: XML文字列が空の場合 

143 xml.etree.ElementTree.ParseError: XMLの解析に失敗した場合 

144 """ 

145 if not xml_str_or_path: 

146 raise ValueError("xml_str_or_path must not be empty") 

147 

148 # ファイルパスかどうかを判定 

149 path = Path(xml_str_or_path) 

150 if path.exists() and path.is_file(): 

151 xml_content = path.read_text(encoding="utf-8") 

152 else: 

153 xml_content = xml_str_or_path 

154 

155 # XMLパース（ParseErrorは自然に伝播させる） 

156 root = ElementTree.fromstring(xml_content) 

157 

158 sections: list[dict[str, Any]] = [] 

159 

160 for dialogue_section in root.iter("dialogue-section"): 

161 section_number = dialogue_section.get("number", "") 

162 section_title = dialogue_section.get("title", "") 

163 

164 # introduction パース 

165 intro_elem = dialogue_section.find("introduction") 

166 if intro_elem is not None: 

167 intro_text = (intro_elem.text or "").strip() 

168 introduction: dict[str, str] | None = { 

169 "speaker": intro_elem.get("speaker", "narrator"), 

170 "text": intro_text, 

171 } 

172 else: 

173 introduction = None 

174 

175 # dialogue/utterance パース 

176 utterances: list[dict[str, str]] = [] 

177 dialogue_elem = dialogue_section.find("dialogue") 

178 if dialogue_elem is not None: 

179 for utterance_elem in dialogue_elem.findall("utterance"): 

180 utterances.append( 

181 { 

182 "speaker": utterance_elem.get("speaker", ""), 

183 "text": (utterance_elem.text or "").strip(), 

184 } 

185 ) 

186 

187 # conclusion パース 

188 conclusion_elem = dialogue_section.find("conclusion") 

189 if conclusion_elem is not None: 

190 conclusion: dict[str, str] | None = { 

191 "speaker": conclusion_elem.get("speaker", "narrator"), 

192 "text": (conclusion_elem.text or "").strip(), 

193 } 

194 else: 

195 conclusion = None 

196 

197 sections.append( 

198 { 

199 "section_number": section_number, 

200 "section_title": section_title, 

201 "introduction": introduction, 

202 "utterances": utterances, 

203 "conclusion": conclusion, 

204 } 

205 ) 

206 

207 return sections

239def synthesize_utterance( 

240 text: str, 

241 speaker_id: str, 

242 synthesizer: Any, 

243 speed_scale: float | None = None, 

244) -> tuple[np.ndarray, int]: 

245 """発話テキストから音声データを生成する. 

246 

247 Args: 

248 text: 発話テキスト 

249 speaker_id: 話者識別子 ("narrator", "A", "B") 

250 synthesizer: VoicevoxSynthesizerインスタンス（またはモック） 

251 speed_scale: 読み上げ速度スケール（Noneの場合はデフォルト） 

252 

253 Returns: 

254 (波形データ as numpy.ndarray, サンプルレート as int) 

255 

256 Raises: 

257 ValueError: text が空文字列の場合 

258 TypeError: text が None の場合 

259 ValueError: 未知の話者IDが指定された場合 

260 """ 

261 if text is None: 

262 raise TypeError("text must not be None") 

263 

264 if not text: 

265 raise ValueError("text must not be empty") 

266 

267 # Apply reading rules and dictionary before TTS 

268 processed_text = apply_readings_to_text(text) 

269 

270 # 話者IDからスタイルIDを取得（未知IDはエラー） 

271 style_id = get_style_id(speaker_id) 

272 

273 # 音声合成 

274 kwargs: dict[str, Any] = {"style_id": style_id} 

275 if speed_scale is not None: 

276 kwargs["speed_scale"] = speed_scale 

277 

278 wav_bytes: bytes = synthesizer.synthesize(processed_text, **kwargs) 

279 

280 # バイト列をnumpy配列に変換 

281 with io.BytesIO(wav_bytes) as buf: 

282 waveform, sample_rate = sf.read(buf) 

283 

284 return np.asarray(waveform, dtype=np.float32), int(sample_rate)

303def concatenate_section_audio( 

304 segments: list[tuple[np.ndarray, int, str]], 

305 silence_duration: float = 0.5, 

306 output_path: Path | str | None = None, 

307) -> tuple[np.ndarray, int]: 

308 """複数の音声セグメントを結合する. 

309 

310 同一話者の連続セグメントは無音なしで結合し、 

311 話者が変わる場合のみ無音を挿入する。 

312 

313 Args: 

314 segments: 音声セグメントのリスト。各要素は (waveform, sample_rate, speaker_id) のタプル。 

315 silence_duration: 話者交代時の無音時間（秒） 

316 output_path: 結合結果の保存先パス。Noneの場合は保存しない。 

317 

318 Returns: 

319 (結合された波形データ as numpy.ndarray, サンプルレート as int) 

320 

321 Raises: 

322 ValueError: segments が空リストの場合 

323 """ 

324 if not segments: 

325 raise ValueError("segments must not be empty") 

326 

327 sample_rate = segments[0][1] 

328 parts: list[np.ndarray] = [] 

329 

330 for i, (waveform, sr, speaker) in enumerate(segments): 

331 parts.append(np.asarray(waveform, dtype=np.float32)) 

332 # 最後のセグメント以外で、次の話者が異なる場合のみ無音を挿入 

333 if i < len(segments) - 1: 

334 next_speaker = segments[i + 1][2] 

335 if speaker != next_speaker: 

336 silence_samples = int(sr * silence_duration) 

337 parts.append(np.zeros(silence_samples, dtype=np.float32)) 

338 

339 combined = np.concatenate(parts) 

340 

341 if output_path is not None: 

342 output_path = Path(output_path) 

343 output_path.parent.mkdir(parents=True, exist_ok=True) 

344 sf.write(str(output_path), combined, sample_rate) 

345 logger.info("Saved combined audio: %s", output_path) 

346 

347 return combined, sample_rate

350def _synthesize_section( 

351 section: dict[str, Any], 

352 synthesizer: Any, 

353 speed_scale: float, 

354) -> list[tuple[np.ndarray, int, str]]: 

355 """1セクションの音声セグメントを生成する. 

356 

357 Args: 

358 section: セクション情報 (parse_dialogue_xml の出力要素) 

359 synthesizer: VoicevoxSynthesizerインスタンス 

360 speed_scale: 読み上げ速度スケール 

361 

362 Returns: 

363 音声セグメントのリスト。各要素は (waveform, sample_rate, speaker_id) のタプル。 

364 """ 

365 section_number = section["section_number"] 

366 section_title = section.get("section_title", "") 

367 logger.info( 

368 "Processing section %s: %s", 

369 section_number or "(no number)", 

370 section_title[:50] + "..." if len(section_title) > 50 else section_title, 

371 ) 

372 

373 segments: list[tuple[np.ndarray, int, str]] = [] 

374 

375 # introduction 

376 intro = section.get("introduction") 

377 if intro and intro.get("text"): 

378 intro_text = intro["text"] 

379 intro_speaker = intro["speaker"] 

380 if is_speakable_text(intro_text): 

381 logger.info( 

382 " [intro] speaker=%s, len=%d: %s", 

383 intro_speaker, 

384 len(intro_text), 

385 intro_text[:80] + "..." if len(intro_text) > 80 else intro_text, 

386 ) 

387 waveform, sr = synthesize_utterance( 

388 text=intro_text, 

389 speaker_id=intro_speaker, 

390 synthesizer=synthesizer, 

391 speed_scale=speed_scale, 

392 ) 

393 segments.append((waveform, sr, intro_speaker)) 

394 else: 

395 logger.warning(" [intro] skipped (no speakable text): %r", intro_text) 

396 

397 # utterances 

398 for i, utterance in enumerate(section.get("utterances", [])): 

399 if utterance.get("text"): 

400 utt_text = utterance["text"] 

401 utt_speaker = utterance["speaker"] 

402 if is_speakable_text(utt_text): 

403 logger.info( 

404 " [utterance %d] speaker=%s, len=%d: %s", 

405 i + 1, 

406 utt_speaker, 

407 len(utt_text), 

408 utt_text[:80] + "..." if len(utt_text) > 80 else utt_text, 

409 ) 

410 waveform, sr = synthesize_utterance( 

411 text=utt_text, 

412 speaker_id=utt_speaker, 

413 synthesizer=synthesizer, 

414 speed_scale=speed_scale, 

415 ) 

416 segments.append((waveform, sr, utt_speaker)) 

417 else: 

418 logger.warning(" [utterance %d] skipped (no speakable text): %r", i + 1, utt_text) 

419 

420 # conclusion 

421 conclusion = section.get("conclusion") 

422 if conclusion and conclusion.get("text"): 

423 concl_text = conclusion["text"] 

424 concl_speaker = conclusion["speaker"] 

425 if is_speakable_text(concl_text): 

426 logger.info( 

427 " [conclusion] speaker=%s, len=%d: %s", 

428 concl_speaker, 

429 len(concl_text), 

430 concl_text[:80] + "..." if len(concl_text) > 80 else concl_text, 

431 ) 

432 waveform, sr = synthesize_utterance( 

433 text=concl_text, 

434 speaker_id=concl_speaker, 

435 synthesizer=synthesizer, 

436 speed_scale=speed_scale, 

437 ) 

438 segments.append((waveform, sr, concl_speaker)) 

439 else: 

440 logger.warning(" [conclusion] skipped (no speakable text): %r", concl_text) 

441 

442 return segments

445def process_dialogue_sections( 

446 sections: list[dict[str, Any]], 

447 synthesizer: Any, 

448 output_dir: Path, 

449 speed_scale: float = 1.0, 

450) -> list[Path]: 

451 """対話セクションリストから音声ファイルを生成する統合関数. 

452 

453 各セクションの introduction, utterances, conclusion を合成し、 

454 チャプター単位でWAVファイルとして保存する。 

455 

456 Args: 

457 sections: parse_dialogue_xml() が返すセクションリスト 

458 synthesizer: VoicevoxSynthesizerインスタンス 

459 output_dir: 音声ファイルの出力ディレクトリ 

460 speed_scale: 読み上げ速度スケール 

461 

462 Returns: 

463 生成したWAVファイルのパスリスト 

464 """ 

465 output_dir = Path(output_dir) 

466 output_dir.mkdir(parents=True, exist_ok=True) 

467 

468 # セクションをチャプター番号でグループ化 

469 chapters: dict[str, list[dict[str, Any]]] = defaultdict(list) 

470 for section in sections: 

471 chapter_num = get_chapter_number(section["section_number"]) 

472 chapters[chapter_num].append(section) 

473 

474 logger.info("Grouped %d sections into %d chapters", len(sections), len(chapters)) 

475 

476 generated: list[Path] = [] 

477 

478 # チャプター番号順にソートして処理 

479 for chapter_num in sorted(chapters.keys(), key=lambda x: (int(x) if x.isdigit() else 0, x)): 

480 chapter_sections = chapters[chapter_num] 

481 logger.info( 

482 "Processing chapter %s (%d sections)", 

483 chapter_num, 

484 len(chapter_sections), 

485 ) 

486 

487 # チャプター内の全セクションの音声セグメントを収集 

488 all_segments: list[tuple[np.ndarray, int, str]] = [] 

489 for section in chapter_sections: 

490 section_segments = _synthesize_section(section, synthesizer, speed_scale) 

491 all_segments.extend(section_segments) 

492 

493 if all_segments: 

494 output_path = output_dir / f"chapter_{chapter_num.zfill(3)}.wav" 

495 concatenate_section_audio(all_segments, output_path=output_path) 

496 generated.append(output_path) 

497 logger.info("Generated chapter audio: %s", output_path) 

498 

499 return generated

572def main() -> int: 

573 """エントリーポイント. 

574 

575 Returns: 

576 終了コード (0: 成功, 1: 入力エラー, 2: VOICEVOX初期化エラー, 3: 音声生成エラー) 

577 """ 

578 

579 logging.basicConfig(level=logging.INFO, format="%(levelname)s: %(message)s") 

580 

581 args = parse_args() 

582 

583 # 入力ファイル確認 

584 input_path = Path(args.input) 

585 if not input_path.exists(): 

586 logger.error("Input file not found: %s", input_path) 

587 return 1 

588 

589 # Initialize reading dictionary from source file 

590 dict_source_path = Path(args.dict_source) if args.dict_source else None 

591 init_readings(dict_source_path) 

592 

593 # Note: カスタムスタイル設定は将来のバージョンで実装予定 

594 # 現在は DEFAULT_SPEAKERS を使用 

595 _ = (args.narrator_style, args.speaker_a_style, args.speaker_b_style) # noqa: F841 

596 

597 try: 

598 # XMLパース 

599 sections = parse_dialogue_xml(str(input_path)) 

600 logger.info("Parsed %d sections", len(sections)) 

601 except Exception as e: 

602 logger.error("Failed to parse dialogue XML: %s", e) 

603 return 1 

604 

605 # ハッシュベースの出力ディレクトリを生成（元のXMLファイルから計算） 

606 if dict_source_path and dict_source_path.exists(): 

607 content_hash = get_xml_content_hash(dict_source_path) 

608 else: 

609 logger.error("--dict-source is required for hash-based output directory") 

610 return 1 

611 output_base = Path(args.output)
