1"""XML2 to TTS pipeline for VOICEVOX (book2.xml format). 

2 

3This module provides a command-line interface to generate TTS audio 

4from book2.xml files using the VOICEVOX synthesizer. 

5 

6Integration with existing components: 

7- xml2_parser: Parse book2.xml and extract content items 

8- text_cleaner: Clean text for TTS consumption 

9- pipeline: Generate audio files 

10- voicevox_client: VOICEVOX synthesis 

11"""

51def parse_args(args: list[str] | None = None) -> argparse.Namespace: 

52 """Parse command line arguments. 

53 

54 Args: 

55 args: List of arguments (for testing). If None, uses sys.argv. 

56 

57 Returns: 

58 argparse.Namespace with parsed arguments 

59 """ 

60 parser = argparse.ArgumentParser(description="Generate TTS audio from book2.xml files") 

61 

62 # Required 

63 parser.add_argument("--input", "-i", required=True, help="Input XML file path") 

64

66 parser.add_argument("--output", "-o", default="./output", help="Output directory (default: ./output)")

68 "--chapter-sound",

78 parser.add_argument("--speed", type=float, default=1.0, help="Speech speed (default: 1.0)")

94def main(args: list[str] | None = None) -> None: 

95 """Main entry point. 

96 

97 Args: 

98 args: List of arguments (for testing). If None, uses sys.argv. 

99 

100 Raises: 

101 FileNotFoundError: If input file does not exist 

102 xml.etree.ElementTree.ParseError: If XML is malformed 

103 """ 

104 parsed = parse_args(args) 

105 

106 # Check file existence 

107 input_path = Path(parsed.input) 

108 if not input_path.exists(): 

109 raise FileNotFoundError(f"Input file not found: {parsed.input}") 

110 

111 # PID file management (Rails-style) 

112 pid_file = get_pid_file_path(str(input_path)) 

113 kill_existing_process(pid_file) 

114 write_pid_file(pid_file) 

115 # Register cleanup on exit 

116 atexit.register(cleanup_pid_file, pid_file) 

117 

118 logger.info("Reading XML: %s", input_path) 

119 

120 # Parse XML (will raise ParseError for invalid XML) 

121 content_items = parse_book2_xml(input_path) 

122 logger.info("Found %d content items in XML", len(content_items)) 

123 

124 if not content_items: 

125 logger.warning("No content items found") 

126 return 

127 

128 # Combine all content text for hash/dict loading 

129 combined_text = " ".join(item.text for item in content_items) 

130 init_for_content(combined_text) 

131 

132 # Generate hash-based output directory 

133 content_hash = get_content_hash(combined_text) 

134 output_base = Path(parsed.output) 

135 output_dir = output_base / content_hash 

136 output_dir.mkdir(parents=True, exist_ok=True) 

137 logger.info("Output directory: %s", output_dir) 

138 

139 # Check if --cleaned-text option is provided 

140 if parsed.cleaned_text: 

141 # Use existing cleaned_text.txt file 

142 cleaned_text_path = Path(parsed.cleaned_text) 

143 if not cleaned_text_path.exists(): 

144 raise FileNotFoundError(f"Cleaned text file not found: {parsed.cleaned_text}") 

145 logger.info("Using existing cleaned text: %s", cleaned_text_path) 

146 else: 

147 # Save cleaned text 

148 cleaned_text_path = output_dir / "cleaned_text.txt" 

149 with open(cleaned_text_path, "w", encoding="utf-8") as f: 

150 current_chapter = None 

151 

152 for item in content_items: 

153 # Insert chapter separator when chapter changes 

154 if item.chapter_number is not None and item.chapter_number != current_chapter: 

155 current_chapter = item.chapter_number 

156 

157 # Find chapter title from heading info 

158 chapter_title = "Untitled" 

159 if item.item_type == "heading" and item.heading_info and item.heading_info.level == 1: 

160 chapter_title = item.heading_info.title 

161 

162 f.write(f"=== Chapter {current_chapter}: {chapter_title} ===\n\n") 

163 

164 # Remove markers before cleaning 

165 text = item.text 

166 if text.startswith(CHAPTER_MARKER): 

167 text = text[len(CHAPTER_MARKER) :] 

168 elif text.startswith(SECTION_MARKER): 

169 text = text[len(SECTION_MARKER) :] 

170 

171 # Apply clean_page_text to remove URLs, parenthetical English, convert numbers, etc. 

172 cleaned = clean_page_text(text) 

173 

174 # For headings, ensure they end with single period (。) 

175 # Remove any trailing punctuation and add one period 

176 if item.item_type == "heading" and cleaned.strip(): 

177 cleaned = cleaned.rstrip() 

178 # Remove all trailing punctuation (、。！？) 

179 while cleaned and cleaned[-1] in "、。！？": 

180 cleaned = cleaned[:-1] 

181 # Add single period at the end 

182 cleaned = cleaned + "。" 

183 

184 # Skip empty content 

185 if cleaned.strip(): 

186 f.write(cleaned) 

187 f.write("\n\n") 

188 

189 logger.info("Saved cleaned text: %s", cleaned_text_path) 

190 

191 # Load sound effects if specified 

192 chapter_sound = None 

193 section_sound = None 

194 

195 if parsed.chapter_sound: 

196 sound_path = Path(parsed.chapter_sound) 

197 if sound_path.exists(): 

198 logger.info("Loading chapter sound: %s", sound_path) 

199 chapter_sound = load_sound(sound_path) 

200 logger.info("Chapter sound loaded (%.2f sec)", len(chapter_sound) / 24000) 

201 else: 

202 logger.warning("Chapter sound file not found: %s", sound_path) 

203 

204 if parsed.section_sound: 

205 sound_path = Path(parsed.section_sound) 

206 if sound_path.exists(): 

207 logger.info("Loading section sound: %s", sound_path) 

208 section_sound = load_sound(sound_path) 

209 logger.info("Section sound loaded (%.2f sec)", len(section_sound) / 24000) 

210 else: 

211 logger.warning("Section sound file not found: %s", sound_path) 

212 

213 # Initialize VOICEVOX synthesizer 

214 voicevox_dir = Path(parsed.voicevox_dir) 

215 config = VoicevoxConfig( 

216 onnxruntime_dir=voicevox_dir / "onnxruntime" / "lib", 

217 open_jtalk_dict_dir=voicevox_dir / "dict" / "open_jtalk_dic_utf_8-1.11", 

218 vvm_dir=voicevox_dir / "models" / "vvms", 

219 style_id=parsed.style_id, 

220 speed_scale=parsed.speed, 

221 pitch_scale=0.0, 

222 volume_scale=1.0, 

223 ) 

224 synthesizer = VoicevoxSynthesizer(config) 

225 synthesizer.initialize() 

226 synthesizer.load_model_for_style_id(parsed.style_id) 

227 logger.info("VOICEVOX initialized (style_id=%d)", parsed.style_id) 

228 

229 # Process content and generate audio 

230 process_chapters( 

231 content_items, 

232 synthesizer, 

233 output_dir, 

234 parsed, 

235 chapter_sound=chapter_sound, 

236 section_sound=section_sound, 

237 ) 

238 logger.info("Audio generation complete") 

239 

240 

241if __name__ == "__main__": 

242 main()
