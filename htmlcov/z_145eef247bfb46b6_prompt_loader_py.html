1"""プロンプトファイルの読み込みユーティリティ。 

2 

3プロンプトファイルを読み込み、プレースホルダーを置換して 

4システムメッセージとユーザーメッセージを返す。 

5 

6プロンプトファイル形式: 

7 [SYSTEM] 

8 システムメッセージ 

9 [USER] 

10 ユーザーメッセージ（{placeholder}形式で置換可能） 

11""" 

12 

13import re 

14from pathlib import Path 

15 

16# プロンプトディレクトリのパス 

17PROMPTS_DIR: Path = Path(__file__).parent / "prompts" 

18 

19 

20class PromptLoadError(Exception): 

21 """プロンプト読み込みエラー。""" 

22 

23 

24def load_prompt(name: str, **kwargs: str) -> tuple[str, str]: 

25 """プロンプトファイルを読み込み、プレースホルダーを置換する。 

26 

27 Args: 

28 name: プロンプト名（拡張子なし） 

29 **kwargs: プレースホルダー置換用のキーワード引数 

30 

31 Returns: 

32 (system_message, user_message) のタプル 

33 

34 Raises: 

35 PromptLoadError: ファイルが見つからない、または形式が不正な場合 

36 

37 Example: 

38 >>> system, user = load_prompt( 

39 ... "generate_dialogue", 

40 ... a_name="教授", 

41 ... b_name="助手", 

42 ... full_context="テキスト内容", 

43 ... ) 

44 """ 

45 prompt_path = PROMPTS_DIR / f"{name}.txt" 

46 

47 if not prompt_path.exists(): 

48 raise PromptLoadError(f"プロンプトファイルが見つかりません: {prompt_path}") 

49 

50 content = prompt_path.read_text(encoding="utf-8") 

51 

52 # [SYSTEM] と [USER] セクションを分割（順序も検証） 

53 if "[SYSTEM]" not in content or "[USER]" not in content: 

54 raise PromptLoadError(f"プロンプトファイルの形式が不正です（[SYSTEM]と[USER]が必要）: {prompt_path}") 

55 

56 if content.index("[SYSTEM]") > content.index("[USER]"): 

57 raise PromptLoadError(f"[SYSTEM]は[USER]より前に配置してください: {prompt_path}") 

58 

59 parts = content.split("[USER]") 

60 if len(parts) != 2: 

61 raise PromptLoadError(f"[USER]セクションが複数あります: {prompt_path}") 

62 

63 system_part = parts[0].replace("[SYSTEM]", "").strip() 

64 user_part = parts[1].strip() 

65 

66 # プレースホルダーを置換 

67 for key, value in kwargs.items(): 

68 placeholder = "{" + key + "}" 

69 system_part = system_part.replace(placeholder, value) 

70 user_part = user_part.replace(placeholder, value) 

71 

72 # 未置換のプレースホルダーをチェック 

73 unreplaced = re.findall(r"\{[a-z_]+\}", system_part + user_part) 

74 if unreplaced: 

75 raise PromptLoadError(f"未置換のプレースホルダーがあります: {unreplaced} in {prompt_path}") 

76 

77 return system_part, user_part

81 """利用可能なプロンプト名のリストを返す。 

82 

83 Returns: 

84 プロンプト名のリスト（拡張子なし） 

85 """ 

86 if not PROMPTS_DIR.exists(): 

87 return [] 

88 

89 return [p.stem for p in PROMPTS_DIR.glob("*.txt")]
