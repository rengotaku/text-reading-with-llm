45def _should_exclude(term: str) -> bool: 

46 """Check if a term should be excluded from technical terms.""" 

47 if term.lower() in STOP_WORDS: 

48 return True 

49 if term.startswith(("www.", "http")): 

50 return True 

51 # Exclude domain-like patterns (e.g., github.com) 

52 domain_suffixes = ("com", "org", "net", "io", "jp", "co", "uk", "us", "dev") 

53 if "." in term and term.split(".")[-1] in domain_suffixes: 

54 return True 

55 if term.startswith("ISBN"): 

56 return True 

57 if re.match(r"^No\.\d", term): 

58 return True 

59 if len(term) <= 2 and term.islower(): 

60 return True 

61 return False

64def extract_technical_terms(text: str) -> list[str]: 

65 """Extract potential technical terms (alphabet-based) from text.""" 

66 # Match sequences of ASCII letters, numbers, and common separators 

67 pattern = r"[A-Za-z][A-Za-z0-9\-_./]*[A-Za-z0-9]|[A-Z]{2,}" 

68 matches = re.findall(pattern, text) 

69 

70 # Deduplicate while preserving order 

71 seen = set() 

72 unique_terms = [] 

73 for term in matches: 

74 normalized = term.strip() 

75 if normalized and normalized not in seen and not _should_exclude(normalized): 

76 seen.add(normalized) 

77 unique_terms.append(normalized) 

78 

79 return unique_terms

82def generate_readings_with_llm( 

83 terms: list[str], 

84 model: str = "gpt-oss:20b", 

85 ollama_chat_func: Any = None, 

86) -> dict[str, str]: 

87 """Generate readings for terms using LLM. 

88 

89 Args: 

90 terms: List of technical terms to convert 

91 model: Ollama model name 

92 ollama_chat_func: Function to call Ollama chat API 

93 

94 Returns: 

95 Dictionary mapping terms to their katakana readings 

96 """ 

97 if not terms: 

98 return {} 

99 

100 # Build prompt 

101 terms_list = "\n".join(f"- {term}" for term in terms) 

102 prompt = f"""以下の技術用語・略語の日本語での読み方をカタカナで答えてください。 

103一般的に使われる読み方で、TTSで自然に聞こえる読みにしてください。 

104 

105用語リスト: 

106{terms_list} 

107 

108JSON形式で出力してください。例: 

109{{"SRE": "エスアールイー", "API": "エーピーアイ", "GitHub": "ギットハブ"}} 

110 

111注意: 

112- 必ずカタカナで出力 

113- 略語は一文字ずつ読む（SRE→エスアールイー） 

114- 固有名詞は一般的な読み方で（Google→グーグル） 

115- 日本語として不自然な場合はそのまま出力しない 

116 

117JSON出力:""" 

118 

119 messages = [ 

120 { 

121 "role": "system", 

122 "content": "あなたは技術用語の読み方を教える専門家です。正確なカタカナ読みを出力してください。", 

123 }, 

124 {"role": "user", "content": prompt}, 

125 ] 

126 

127 if ollama_chat_func is None: 

128 raise ValueError("ollama_chat_func must be provided") 

129 

130 response = ollama_chat_func(model=model, messages=messages) 

131 

132 # Parse JSON response 

133 try: 

134 # Extract JSON from response 

135 response_text = response.get("message", {}).get("content", "") 

136 # Try to find JSON in the response 

137 json_match = re.search(r"\{[^{}]*\}", response_text, re.DOTALL) 

138 if json_match: 

139 readings = json.loads(json_match.group()) 

140 return {k: v for k, v in readings.items() if isinstance(v, str)} 

141 except (json.JSONDecodeError, KeyError) as e: 

142 logger.warning("Failed to parse LLM response: %s", e) 

143 

144 return {}

156 """Save reading dictionary to file."""

163def apply_llm_readings(text: str, readings: dict[str, str]) -> str: 

164 """Apply LLM-generated readings to text.""" 

165 # Sort by length (longest first) to avoid partial replacements 

166 sorted_terms = sorted(readings.keys(), key=len, reverse=True) 

167 

168 for term in sorted_terms: 

169 reading = readings[term] 

170 # Use word boundary-like pattern for ASCII terms 

171 pattern = rf"(?<![A-Za-z]){re.escape(term)}(?![A-Za-z])" 

172 text = re.sub(pattern, reading, text) 

173 

174 return text
